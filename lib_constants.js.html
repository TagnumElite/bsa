'use strict'

/* -------------- Default header data -------------- */

/** 
 * Magic for Morrowind BSA.
 * @const {Number}
 * @default 256
 */
exports.MW_BSAHEADER_FILEID = 0x00000100
/** 
 * Magic for Oblivion BSA, the literal string "BSA\0".
 * @const {Number}
 * @default 4281154
 */
exports.OB_BSAHEADER_FILEID = 0x00415342
/** 
 * Magic for Fallout 4 BA2, the literal string "BTDX".
 * @const {Number}
 * @default 1480873026
 */
exports.F4_BSAHEADER_FILEID = 0x58445442

/**
 * Version number of an Oblivion BSA.
 * @const {Number}
 * @default 103
 */
exports.OB_BSAHEADER_VERSION = 0x67
/**
 * Version number of a Fallout 3 BSA.
 * @const {Number}
 * @default 104
 */
exports.F3_BSAHEADER_VERSION = 0x68
/**
 * Version number of a Skyrim SE BSA.
 * @const {Number}
 * @default 105
 */
exports.SSE_BSAHEADER_VERSION = 0x69
/**
 * Version number of a Fallout 4 BA2.
 * @const {Number}
 * @default 1
 */
exports.F4_BSAHEADER_VERSION = 0x01


/* ---------------- Archive flags ---------------- */

/**
 * Whether the BSA has names for paths.
 * @const {Number}
 * @default 1
 */
exports.OB_BSAARCHIVE_PATHNAMES = 0x0001
/**
 * Whether the BSA has names for files.
 * @const {Number}
 * @default 2
 */
exports.OB_BSAARCHIVE_FILENAMES = 0x0002
/**
 * Whether the files are compressed.
 * @const {Number}
 * @default 4
 */
exports.OB_BSAARCHIVE_COMPRESSFILES = 0x0004
/**
 * Whether the name is prefixed to the data?
 * @const {Number}
 * @default 256
 */
exports.F3_BSAARCHIVE_PREFIXFULLFILENAMES = 0x0100


/* ------------------ File flags ------------------ */

/** Set when the BSA contains NIF files.
 * @const {Number}
 * @default 1
 */
exports.OB_BSAFILE_NIF = 0x0001
/** Set when the BSA contains DDS files.
 * @const {Number}
 * @default 2
 */
exports.OB_BSAFILE_DDS = 0x0002
/** Set when the BSA contains XML files.
 * @const {Number}
 * @default 4
 */
exports.OB_BSAFILE_XML = 0x0004
/** Set when the BSA contains WAV files.
 * @const {Number}
 * @default 8
 */
exports.OB_BSAFILE_WAV = 0x0008
/** Set when the BSA contains MP3 files.
 * @const {Number}
 * @default 16
 */
exports.OB_BSAFILE_MP3 = 0x0010
/** Set when the BSA contains TXT files.
 * @const {Number}
 * @default 32
 */
exports.OB_BSAFILE_TXT = 0x0020
/** Set when the BSA contains HTML files.
 * @const {Number}
 * @default 32
 */
exports.OB_BSAFILE_HTML = 0x0020
/** Set when the BSA contains BAT files.
 * @const {Number}
 * @default 32
 */
exports.OB_BSAFILE_BAT = 0x0020
/** Set when the BSA contains SCC files.
 * @const {Number}
 * @default 32
 */
exports.OB_BSAFILE_SCC = 0x0020
/** Set when the BSA contains SPT files.
 * @const {Number}
 * @default 64
 */
exports.OB_BSAFILE_SPT = 0x0040
/** Set when the BSA contains TEX files.
 * @const {Number}
 * @default 128
 */
exports.OB_BSAFILE_TEX = 0x0080
/** Set when the BSA contains FNT files.
 * @const {Number}
 * @default 128
 */
exports.OB_BSAFILE_FNT = 0x0080
/** Set when the BSA contains CTL files.
 * @const {Number}
 * @default 256
 */
exports.OB_BSAFILE_CTL = 0x0100


/* -- Bitmasks for the size field in the header -- */

/** Bit mask to get the size of the file.
 * @const {Number}
 * @default 1073741823
 */
exports.OB_BSAFILE_SIZEMASK = 0x3fffffff


/* -------------- Record flags -------------- */

/** Bit mask to get the compression status.
 * @const {Number}
 * @default 3221225472
 */
exports.OB_BSAFILE_FLAG_COMPRESS = 0xC0000000
